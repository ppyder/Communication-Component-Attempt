114 bool InputFilter(StateFilter *pFilter, bool NowState)
115 {
116  uint32_t temp = 0;
117 
118  if(pFilter == NULL)
119  return false;
120 
121  NowState &= FILTER_MASK_CODE[1]; //取末位，将非法输入的危害降到最小
122 
123  pFilter->History <<= 1;
124  pFilter->History |= NowState; //更新历史
125 
126  //取最新的 滤波位数 位
127  temp = (pFilter->History) & FILTER_MASK_CODE[pFilter->FilteringBits];
128 
129  if(temp == FILTER_MASK_CODE[pFilter->FilteringBits]) //被关注的位都是1，说明进入所谓“高电平状态”
130  {
131  if(pFilter->State != true)
132  {
133  pFilter->Jumping = RISING_EDGE; //标记触发上升沿
134  pFilter->State = true;
135  }
136  }
137  else if(temp == FILTER_MASK_CODE[0]) //进入“低电平状态”
138  {
139  if(pFilter->State != false)
140  {
141  pFilter->Jumping = RISING_EDGE; //标记触发下降沿
142  pFilter->State = false;
143  }
144  }
145 
146  return pFilter->State;
147 }
